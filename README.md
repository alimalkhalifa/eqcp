SELECT * FROM spawnentry se 
INNER JOIN spawn2 s2 ON s2.`spawngroupid` = se.`spawngroupID` 
INNER JOIN npc_types n ON n.id = se.npcid
WHERE s2.enabled = 1 AND s2.zone = "qeynos"
